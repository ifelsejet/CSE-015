\section*{Exercise 1}
Solve $(x-2)(x+4) = 7$
\begin{align*}
x^2 + 4x -2x - 8 - 7 = 0\\
x^2 +2x -15 = 0\\
(x-3)(x+5) = 0
\end{align*}
Therefore $x=3$ or $x=-5$

\section*{Exercise 2}
Solve $(x-3)^2 = 4$
\begin{align*}
\sqrt{x-3}^2 = \sqrt{4}\\
(x-3) = \pm2\\
x-3 = 2 \\
x = 5 \\
x-3 = -2 \\
x = 1
\end{align*}
Therefore $x=1$ or $x=5$

\section*{Exercise 3}
Solve $\sqrt{x-2} + 2 = 4$
\begin{align*}
\sqrt{x-2} = 2 \\
\sqrt{x-2}^2 = 2^2 \\
x-2 = 4 \\
x = 6
\end{align*}
Therefore $x=6$
